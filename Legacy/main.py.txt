class DriveManager:
    """Оновлений клас з авторизацією через OAuth (від імені людини)"""

    """Клас для роботи з Google Drive/Docs (Авторизація + PDF)"""

    def __init__(self):
        self.creds = None
        if os.path.exists('token.json'):
            self.creds = Credentials.from_authorized_user_file('token.json', SCOPES)
        if not self.creds or not self.creds.valid:
            if self.creds and self.creds.expired and self.creds.refresh_token:
                self.creds.refresh(Request())
            else:
                # Увага: При першому запуску на сервері це може вимагати ручного втручання
                # Краще згенерувати token.json локально і залити на сервер
                flow = InstalledAppFlow.from_client_secrets_file(CLIENT_SECRET_FILE, SCOPES)
                self.creds = flow.run_local_server(port=0)
            with open('token.json', 'w') as token:
                token.write(self.creds.to_json())

        self.drive_service = build('drive', 'v3', credentials=self.creds)
        self.docs_service = build('docs', 'v1', credentials=self.creds)


    def get_available_templates(self):
        results = self.drive_service.files().list(
            q=f"'{TEMPLATES_FOLDER_ID}' in parents and mimeType='application/vnd.google-apps.document' and trashed=false",
            fields="files(id, name)"
        ).execute()
        return results.get('files', [])


    def create_pdf_from_template(self, template_id, template_name, data):
        # --- БЛОК АВТОМАТИЧНИХ ДАНИХ ---
        # 1. Дата підписання - завжди сьогодні
        data['date_of_signing'] = datetime.date.today().strftime("%d.%m.%Y")

        # 2. Якщо ШІ не вирахував курс, але є група - спробуємо "хак" (для КІПЗ-25 тощо)
        # Це опціонально, але корисно для ваших груп
        if 'group' in data and ('study_year' not in data or not data['study_year']):
            import re
            # Шукаємо дві цифри у назві групи (наприклад 25 з КІПЗ-25)
            match = re.search(r'-(\d{2})', data['group'])
            if match:
                entry_year = 2000 + int(match.group(1))
                current_year = datetime.date.today().year
                current_month = datetime.date.today().month
                # Якщо зараз вересень-грудень, то курс = поточному - вступ + 1
                # Якщо січень-червень, то курс = поточному - вступ
                course = current_year - entry_year + (1 if current_month >= 9 else 0)
                if course > 0:
                    data['study_year'] = str(course)

        # --- СТАНДАРТНА ЛОГІКА ---
        doc_id = None
        try:
            copy_body = {'name': f"TEMP_{template_name}", 'parents': [TEMPLATES_FOLDER_ID]}
            file_copy = self.drive_service.files().copy(fileId=template_id, body=copy_body).execute()
            doc_id = file_copy['id']

            requests = []
            for key, value in data.items():
                safe_value = str(value) if value is not None else ""

                # ТУТ ГОЛОВНА МАГІЯ:
                # Ми перебираємо всі можливі варіанти написання змінної в шаблоні:
                # {{key}}, {{KEY}}, {{Key}}
                # Це гарантує, що DATE_FROM заповниться навіть якщо прийшло date_from

                variants = [key.lower(), key.upper(), key.capitalize()]

                for var in variants:
                    requests.append({
                        'replaceAllText': {
                            'containsText': {'text': f'{{{{{var}}}}}', 'matchCase': True},
                            # matchCase True, бо ми самі перебираємо варіанти
                            'replaceText': safe_value
                        }
                    })

            if requests:
                self.docs_service.documents().batchUpdate(documentId=doc_id, body={'requests': requests}).execute()

            request = self.drive_service.files().export_media(fileId=doc_id, mimeType='application/pdf')
            fh = io.BytesIO()
            downloader = MediaIoBaseDownload(fh, request)
            done = False
            while not done:
                status, done = downloader.next_chunk()

            return fh.getvalue()

        except Exception as e:
            print(f"⚠️ API Error: {e}")
            raise e
        finally:
            if doc_id:
                try:
                    self.drive_service.files().delete(fileId=doc_id).execute()
                except:
                    pass


class GeminiBrain:
    def __init__(self):
        self.client = genai.Client(api_key=GEMINI_API_KEY)
        self.model = GEMINI_AI_MODEL

    def analyze_dialog_turn(self, history, template_list, known_data=None, current_active_template=None):
        templates_str = "\n".join([f"- {t['name']}" for t in template_list])
        current_date_obj = datetime.date.today()
        current_date_str = current_date_obj.strftime("%d.%m.%Y")

        # Визначаємо поточний семестр для промпта
        current_semester = "1" if current_date_obj.month >= 8 or current_date_obj.month == 1 else "2"

        # Логіка базового року для обчислення курсу
        if current_date_obj.month >= 8:
            academic_year_base = current_date_obj.year
        else:
            academic_year_base = current_date_obj.year - 1

        known_data_str = json.dumps(known_data, ensure_ascii=False, indent=2)
        chat_context = "\n".join([f"{msg['role']}: {msg['content']}" for msg in history])

        # Формуємо чітку інструкцію, якщо шаблон вже обрано
        focus_instruction = ""
        if current_active_template:
            focus_instruction = f"URGENT: Active template is '{current_active_template}'. Focus ONLY on gathering missing fields for this specific document."

        response_schema = {
            "type": "OBJECT",
            "properties": {
                "status": {
                    "type": "STRING",
                    "enum": ["CLARIFICATION_NEEDED", "TEMPLATE_SELECTED", "WAITING_FOR_CONFIRMATION",
                             "READY_TO_GENERATE"]
                },
                "bot_reply": {"type": "STRING"},
                "selected_template_name": {"type": "STRING", "nullable": True},
                "extracted_data": {
                    "type": "OBJECT",
                    "properties": {
                        "STUDENTS_NAME": {"type": "STRING"},
                        "STUDENTS_EMAIL": {"type": "STRING"},
                        "DATE_OF_BIRTH": {"type": "STRING"},
                        "SPECIALITY": {"type": "STRING"},
                        "GROUP": {"type": "STRING"},
                        "STUDENTS_NUMBER": {"type": "STRING"},
                        "PARENTS_NAME": {"type": "STRING"},
                        "PARENTS_NUMBER": {"type": "STRING"},
                        "DATE_FROM": {"type": "STRING"},
                        "DATE_TO": {"type": "STRING"},
                        "REASON": {"type": "STRING"},
                        "SPECIALITY_TO": {"type": "STRING"},
                        "SUBJECT": {"type": "STRING"},
                        "DATE_OF_SIGNING": {"type": "STRING"},
                        "STUDY_YEAR": {"type": "STRING"},
                        "STUDY_SEMESTER": {"type": "STRING"},
                        "LESSONS_RANGE": {"type": "STRING"},
                    },
                    "nullable": True
                }
            },
            "required": ["status", "bot_reply"]
        }

        prompt = f"""
You are the AI Administrator for UKD College (Фаховий коледж УКД).
GOAL: Identify the student's request, gather necessary data, and output JSON to generate a PDF document.

### 1. SYSTEM CONTEXT
- **Current Date:** {current_date_str}
- **Academic Year Base:** {academic_year_base} (Used for course calculation)
- **Current Semester:** {current_semester}
- **Available Templates:**
{templates_str}
{focus_instruction}

### 2. STUDENT DATA (DB PRIORITY)
{known_data_str}

### 3. RULES FOR PROFILE UPDATES (HIGHEST PRIORITY)
If the user wants to change their info (e.g., "I want to change my group", "New phone number"):
- ALLOWED fields: "GROUP", "SPECIALITY", "STUDENTS_NUMBER", "PARENTS_NUMBER".
- PROHIBITED fields: "STUDENTS_NAME", "EMAIL" (Reply: "Names and Emails can only be changed by the College Administration").
- ACTION:
  - If field is allowed and new value is provided -> Output status `PROFILE_UPDATE`. Fill `extracted_data` with ONE key-value pair (e.g., "GROUP": "New Name").
  - If field is allowed but value is missing -> Output `CLARIFICATION_NEEDED` (ask for the value).


### 4. DATA PROCESSING RULES
**Priority:** Database > Chat History > New User Input.
**Format Enforcement:**
- **STUDENTS_NAME:** Full Cyrillic Name (Surname Name Patronymic), Title Case (e.g., "Шевченко Тарас Григорович").
- **GROUP:** Format "PrefixSpecForm-Year-Num" (e.g., "КІПЗс-25-1").
- **DATE_OF_BIRTH / DATE_FROM / DATE_TO:** STRICTLY "DD.MM.YYYY". Convert relative dates ("next monday") to exact dates.
- **EMAIL:** Must contain "@ukd.edu.ua".
- **REASON:** Concise linguistic construction (e.g., "сімейними обставинами" not "сімейні обставини", "поїздкою за кордон" not "поїздка за кордон" etc).

**Auto-Calculated Fields (DO NOT ASK USER):**
- **DATE_OF_SIGNING:** Always "{current_date_str}".
- **STUDY_YEAR:** Calculate as: ({academic_year_base} - [Year Digits from Group Name]) + 1. 
  *(Example: Group '...-25-...' means entrance 2025. If Base is 2026, Year = 2).*
- **STUDY_SEMESTER:** Always "{current_semester}".

### 5. WORKFLOW LOGIC (STRICT ORDER)
1. **Identify Template:** If unclear -> Ask. If clear -> `TEMPLATE_SELECTED`.
2. **Gather Data:** Check if ALL required fields for the template are present.
   - If missing info -> `CLARIFICATION_NEEDED`. Ask specific questions.
3. **VERIFICATION (CRITICAL STEP):** - Once ALL data is gathered (manual + auto), **DO NOT** generate yet.
   - Set status to `WAITING_FOR_CONFIRMATION`.
   - In `bot_reply`, explicitly list all collected data (Name, Dates, Reason, etc.) and ask the student: "Чи все вірно? (Yes/No)".
4. **Finalize:** - IF status was `WAITING_FOR_CONFIRMATION` AND user says "Yes/Confirm/Ok/Вірно" -> `READY_TO_GENERATE`.
   - IF user corrects data -> Go back to step 2 (Update data and Verify again).

### 5. CHAT HISTORY
{chat_context}

Respond strictly in JSON.
"""


        response = self.client.models.generate_content(
            model=self.model,
            contents=prompt,
            config=types.GenerateContentConfig(
                response_mime_type="application/json",
                response_schema=response_schema
            )
        )
        try:
            # --- ФІКС: ОЧИСТКА ВІД MARKDOWN ---
            # Gemini іноді пхає ```json ... ``` навіть в JSON mode
            raw_text = response.text.strip()
            if raw_text.startswith("```"):
                raw_text = raw_text.strip("`").replace("json\n", "", 1).strip()

            return json.loads(raw_text)

        except json.JSONDecodeError as e:
            print(f"❌ JSON PARSING ERROR!\nRaw Response: {response.text}")  # Щоб ви бачили в консолі
            return {
                "status": "CLARIFICATION_NEEDED",
                "bot_reply": "Виникла технічна помилка обробки. Спробуйте підтвердити ще раз."
            }
        except Exception as e:
            print(f"❌ UNKNOWN ERROR: {e}")
            return {
                "status": "CLARIFICATION_NEEDED",
                "bot_reply": "Щось пішло не так. Повторіть запит."
            }


class SheetManager:
    def __init__(self, credentials, spreadsheet_id):
        self.service = build('sheets', 'v4', credentials=credentials)
        self.spreadsheet_id = spreadsheet_id

    def get_student_data(self, telegram_id):
        """Шукає студента за Telegram ID (старий метод)"""
        try:
            result = self.service.spreadsheets().values().get(
                spreadsheetId=self.spreadsheet_id, range="Students!A:Z"
            ).execute()
            rows = result.get('values', [])
            if not rows: return {}

            headers = rows[0]
            try:
                tg_id_idx = next(i for i, h in enumerate(headers) if h.strip() == 'Telegram_ID')
            except StopIteration:
                return {}

            for row in rows[1:]:
                if len(row) > tg_id_idx and str(row[tg_id_idx]).strip() == str(telegram_id):
                    student_data = {}
                    for i, header in enumerate(headers):
                        val = row[i] if i < len(row) else ""
                        if val.strip():
                            student_data[header.strip()] = val.strip()
                    return student_data
            return {}
        except Exception as e:
            print(f"Google Sheets API Error: {e}")
            return {}

    def get_all_students(self):
        """Повертає список всіх студентів (допоміжний метод)"""
        try:
            result = self.service.spreadsheets().values().get(
                spreadsheetId=self.spreadsheet_id, range="Students!A:Z"
            ).execute()
            rows = result.get('values', [])

            if len(rows) < 2: return []

            headers = [h.strip() for h in rows[0]]
            students_list = []

            for row in rows[1:]:
                student_data = {}
                for i, header in enumerate(headers):
                    val = row[i] if i < len(row) else ""
                    student_data[header] = val
                students_list.append(student_data)

            return students_list
        except Exception as e:
            print(f"Error fetching all students: {e}")
            return []

    def get_student_by_email(self, email):
        """Шукає студента за email (для авторизації)"""
        try:
            # Викликаємо внутрішній метод класу через self.
            all_students = self.get_all_students()

            target_email = email.strip().lower()

            for student in all_students:
                raw_email = str(student.get('STUDENTS_EMAIL', ''))
                student_email = raw_email.strip().lower()

                if student_email == target_email:
                    return student
            return None
        except Exception as e:
            print(f"Error searching by email: {e}")  # Замінив logging на print для простоти, або використовуйте logging
            return None

    def link_telegram_id(self, email, telegram_id):
        """Записує Telegram ID у таблицю для знайденого студента"""
        try:
            # 1. Завантажуємо всю таблицю, щоб знайти номер рядка
            result = self.service.spreadsheets().values().get(
                spreadsheetId=self.spreadsheet_id, range="Students!A:Z"
            ).execute()
            rows = result.get('values', [])

            headers = [h.strip() for h in rows[0]]

            # Шукаємо індекси колонок
            try:
                email_col_idx = next(i for i, h in enumerate(headers) if h.lower() in ['email', 'пошта'])
                tg_col_idx = next(i for i, h in enumerate(headers) if h == 'Telegram_ID')
            except StopIteration:
                print("Error: Не знайдено колонку 'Email' або 'Telegram_ID'")
                return False

            target_email = email.strip().lower()
            row_number = -1

            # 2. Шукаємо рядок студента
            for i, row in enumerate(rows):
                if i == 0: continue  # Пропускаємо заголовок

                curr_email = str(row[email_col_idx] if len(row) > email_col_idx else "").strip().lower()

                if curr_email == target_email:
                    row_number = i + 1  # +1 бо в Sheets нумерація з 1
                    break

            if row_number == -1: return False

            # 3. Формуємо адресу комірки (наприклад, G5)
            # Конвертуємо індекс колонки в літеру (0->A, 1->B...)
            col_letter = chr(ord('A') + tg_col_idx)
            # (Примітка: це спрацює для колонок A-Z. Якщо колонок більше 26, треба складнішу логіку,
            # але для студентської бази зазвичай вистачає).

            cell_range = f"Students!{col_letter}{row_number}"

            # 4. Записуємо значення
            body = {'values': [[str(telegram_id)]]}
            self.service.spreadsheets().values().update(
                spreadsheetId=self.spreadsheet_id,
                range=cell_range,
                valueInputOption="RAW",
                body=body
            ).execute()

            print(f"✅ Telegram ID {telegram_id} прив'язано до {email}")
            return True

        except Exception as e:
            print(f"Error linking Telegram ID: {e}")
            return False


class EmailManager:
    def __init__(self, credentials):
        # Підключаємося до Gmail API
        self.service = build('gmail', 'v1', credentials=credentials)

    def send_email(self, to_email, subject, body, file_bytes_io, filename):
        """
        Відправляє лист з вкладенням.
        file_bytes_io: об'єкт BytesIO з PDF-файлом
        """
        try:
            # Створюємо повідомлення
            message = MIMEMultipart()
            message['to'] = to_email
            message['subject'] = subject

            # Тіло листа
            message.attach(MIMEText(body, 'plain'))

            # Вкладення (PDF)
            file_bytes_io.seek(0)  # Переконуємось, що курсор на початку
            part = MIMEApplication(file_bytes_io.read(), Name=filename)
            part['Content-Disposition'] = f'attachment; filename="{filename}"'
            message.attach(part)

            # Повертаємо курсор назад, щоб інші функції теж могли читати файл
            file_bytes_io.seek(0)

            # Кодування для Gmail API
            raw_message = base64.urlsafe_b64encode(message.as_bytes()).decode()
            create_message = {'raw': raw_message}

            # Відправка
            self.service.users().messages().send(userId="me", body=create_message).execute()
            print(f"📧 Лист успішно відправлено на {to_email}")
            return True
        except Exception as e:
            print(f"❌ Помилка відправки пошти: {e}")
            return False

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    Точка входу.
    1. Перевіряє RAM.
    2. Перевіряє БД (Telegram_ID).
    3. Якщо не знайдено — просимо Email.
    """
    user_id = str(update.effective_user.id)
    chat_id = update.effective_chat.id

    reset_timeout_timer(user_id, chat_id, context)

    # --- ЕТАП 1: Перевірка оперативної пам'яті (RAM) ---
    if user_id in user_sessions and user_sessions[user_id].get('profile'):
        student_name = user_sessions[user_id]['profile'].get('STUDENTS_NAME', 'Студенте')
        await update.message.reply_text(
            f"👋 З поверненням, {student_name}!\n"
            "Ви вже авторизовані. Можете формувати заяви."
        )
        return

    # --- ЕТАП 2: Перевірка бази даних (Google Sheets) ---
    # Це виправить проблему після перезапуску бота
    print(f"🔄 /start: Перевіряю {user_id} в базі...")
    student_from_db = sheet_mgr.get_student_data(user_id)

    if student_from_db:
        # Ура, знайшли в базі! Відновлюємо сесію
        user_sessions[user_id] = {
            'history': [],
            'profile': student_from_db,
            'active_template': None
        }
        student_name = student_from_db.get('STUDENTS_NAME', 'Студенте')

        await update.message.reply_text(
            f"👋 З поверненням, {student_name}!\n"
            "Чим можу допомогти цього разу?"
        )
        return

    # --- ЕТАП 3: Реєстрація (Якщо ніде не знайдено) ---
    # Створюємо пусту сесію
    user_sessions[user_id] = {'history': [], 'profile': {}, 'active_template': None}

    await update.message.reply_text(
        "👋 Вітаю! Я ШІ-адміністратор Фахового коледжу УКД.\n\n"
        "Для початку роботи мені потрібно верифікувати вас як студента.\n"
        "📧 Будь ласка, **напишіть вашу корпоративну пошту** (наприклад: *ivanov.i@ukd.edu.ua*)."
    , parse_mode="Markdown")

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = str(update.effective_user.id)
    chat_id = update.effective_chat.id
    text = update.message.text.strip()

    # Скидаємо таймер
    reset_timeout_timer(user_id, chat_id, context)

    # Ініціалізація змінної сесії
    if user_id not in user_sessions:
        user_sessions[user_id] = {'history': [], 'profile': {}, 'active_template': None}

    session = user_sessions[user_id]

    # ==========================================
    # 🔒 БЛОК 1: АВТОРИЗАЦІЯ
    # ==========================================

    # 1. Якщо профіль порожній -> Пробуємо "тиху" авторизацію через БД (якщо перезапустили бота)
    if not session.get('profile'):
        print(f"🔄 Спроба знайти користувача {user_id} в базі за ID...")
        student_from_db = sheet_mgr.get_student_data(user_id)

        if student_from_db:
            # УРА! Знайшли. Відновлюємо сесію мовчки.
            session['profile'] = student_from_db
            print(f"✅ Відновлено сесію для {student_from_db.get('STUDENTS_NAME')}")
            # Якщо користувач просто написав /start або привіт, кажемо "З поверненням"
            if text == "/start":
                await update.message.reply_text(f"👋 З поверненням, {student_from_db.get('STUDENTS_NAME')}! Чим можу допомогти?")
                return

        else:
            # 2. Якщо в БД немає -> Це дійсно новий користувач або не прив'язаний.
            # Починаємо процедуру реєстрації через Email.

            if "@ukd.edu.ua" not in text.lower():
                await update.message.reply_text(
                    "🔒 Необхідна авторизація.\n"
                    "Введіть вашу корпоративну пошту (**... @ukd.edu.ua**), щоб я міг вас запам'ятати."
                    , parse_mode="Markdown")
                return

            msg = await update.message.reply_text(f"🔎 Перевіряю пошту {text}...")
            student = sheet_mgr.get_student_by_email(text)

            if student:
                # Знайшли за поштою!

                # --- ВАЖЛИВО: ЗАПИСУЄМО TELEGRAM_ID В БАЗУ ---
                success = sheet_mgr.link_telegram_id(text, user_id)

                session['profile'] = student
                student_name = student.get('STUDENTS_NAME', 'Студенте')

                await context.bot.delete_message(chat_id=chat_id, message_id=msg.message_id)

                welcome_text = f"✅ Авторизація успішна!\nВітаю, {student_name}."
                if success:
                    welcome_text += "\n💾 Я запам'ятав вас, наступного разу вхід буде автоматичним."

                await update.message.reply_text(welcome_text)
                await update.message.reply_text("Яку заяву ви хочете оформити?")
            else:
                await context.bot.edit_message_text(
                    chat_id=chat_id,
                    message_id=msg.message_id,
                    text="⛔ Пошта не знайдена в базі. Перевірте правильність або зверніться в деканат."
                )
            return

    # ==========================================
    # 🧠 БЛОК 2: ШІ ТА ГЕНЕРАЦІЯ
    # ==========================================
    # (Виконується лише для авторизованих користувачів)

    # 1. Додаємо повідомлення в історію
    session['history'].append({"role": "user", "content": text})

    await context.bot.send_chat_action(chat_id=chat_id, action="typing")

    templates = drive_mgr.get_available_templates()

    try:
        analysis = brain.analyze_dialog_turn(
            history=session['history'],
            template_list=templates,
            known_data=session['profile'],
            current_active_template=session['active_template']
        )
    except Exception as e:
        await update.message.reply_text("😵 Виникла помилка при обробці ШІ. Спробуйте ще раз.")
        logging.error(f"Brain Error: {e}")
        return

    status = analysis.get("status")
    reply = analysis.get("bot_reply")
    data = analysis.get("extracted_data", {})

    print(f"Status: {status} | Template: {analysis.get('selected_template_name')}")

    # Зберігаємо відповідь моделі
    session['history'].append({"role": "model", "content": reply})

    # --- СЦЕНАРІЙ А: УТОЧНЕННЯ ---
    if status in ["CLARIFICATION_NEEDED", "TEMPLATE_SELECTED", "WAITING_FOR_CONFIRMATION"]:
        await update.message.reply_text(reply)
        if status == "TEMPLATE_SELECTED":
            session['active_template'] = analysis.get("selected_template_name")
        return

    # --- СЦЕНАРІЙ Б: ГЕНЕРАЦІЯ ---
    elif status == "READY_TO_GENERATE":

        # 1. Підстраховка: якщо Gemini забув назву шаблону в JSON, беремо з пам'яті
        current_tmpl_name = session.get('active_template')  # Беремо з сесії

        # Якщо і там немає, пробуємо взяти з відповіді Gemini
        if not current_tmpl_name:
            candidate = analysis.get("selected_template_name")
            if candidate:
                current_tmpl_name = candidate

        # Дебаг-принт, щоб бачити, що відбувається
        print(f"🏁 Generation Step. Template: {current_tmpl_name}")

        if not current_tmpl_name:
            await update.message.reply_text(
                "⚠️ Все готово, але я загубив назву шаблону. Нагадайте, яку заяву друкуємо?")
            return

        tmpl_obj = next((t for t in templates if t['name'] == current_tmpl_name), None)
        if not tmpl_obj:
            await update.message.reply_text("❌ Помилка: файл шаблону не знайдено.")
            return

        # Початок процесу (Message Editing)
        status_msg = await update.message.reply_text("⏳ Генерую документ...")

        try:
            # 1. ГЕНЕРАЦІЯ PDF
            await context.bot.send_chat_action(chat_id=chat_id, action="upload_document")

            raw_pdf = drive_mgr.create_pdf_from_template(tmpl_obj['id'], current_tmpl_name, data)

            if isinstance(raw_pdf, bytes):
                pdf_file = io.BytesIO(raw_pdf)
            else:
                pdf_file = raw_pdf

            student_name = data.get('STUDENTS_NAME', 'Student')
            clean_name = student_name.replace(' ', '_')
            filename = f"Заява_{clean_name}.pdf"
            pdf_file.name = filename

            # 2. EMAIL
            await context.bot.edit_message_text(
                chat_id=chat_id,
                message_id=status_msg.message_id,
                text="📤 Відправляю на пошту..."
            )

            email_subject = f"ДРУК: Заява - {student_name} ({data.get('GROUP', '')})"
            email_body = f"""
            Доброго дня.
            Надійшла нова заява від студента для друку.

            Студент: {student_name}
            Група: {data.get('GROUP', 'Не вказано')}
            Тип заяви: {current_tmpl_name}
            """

            pdf_file.seek(0)  # Скидаємо курсор

            email_mgr.send_email(
                to_email=TARGET_PRINT_EMAIL,
                subject=email_subject,
                body=email_body,
                file_bytes_io=pdf_file,
                filename=filename
            )

            # 3. ФІНАЛ
            await context.bot.edit_message_text(
                chat_id=chat_id,
                message_id=status_msg.message_id,
                text="✅ Заяву сформовано та відправлено на друк!\nМожете забрати її в 300 кабінеті."
            )

            pdf_file.seek(0)  # Знову скидаємо курсор для Telegram

            await update.message.reply_document(
                document=pdf_file,
                filename=filename,
                caption="Ваша копія заяви 📄"
            )

            # ОЧИЩЕННЯ
            session['history'] = []
            session['active_template'] = None
            # session['profile'] = {} <-- ЦЕЙ РЯДОК Я ПРИБРАВ/ЗАКОМЕНТУВАВ
            # Ми залишаємо профіль заповненим, щоб студенту не треба було вводити пошту знову.

            await update.message.reply_text("Готовий до нових завдань. Щось ще?")

        except Exception as e:
            logging.error(f"Generation Error: {e}")
            await context.bot.edit_message_text(
                chat_id=chat_id,
                message_id=status_msg.message_id,
                text=f"❌ Сталася помилка: {e}"
            )
    else:
        # Це виправить "мовчання" бота.
        # Якщо статус невідомий (наприклад, GENERAL_TALK або помилка моделі),
        # але модель згенерувала текст відповіді - ми маємо його показати.
        print(f"⚠️ Нестандартний статус: {status}")  # Для логів

        if reply:
            await update.message.reply_text(reply)
        else:
            # Якщо і статус дивний, і відповіді немає
            await update.message.reply_text(
                "🤔 Я почув вас, але не впевнений, що робити далі. Спробуйте перефразувати запит.")


async def session_timeout_callback(context: ContextTypes.DEFAULT_TYPE):
    """Ця функція запускається автоматично через 1 годину бездіяльності"""
    job = context.job
    user_id = job.data  # Ми передали ID користувача в job.data
    chat_id = job.chat_id

    # Перевіряємо, чи є ще сесія
    if user_id in user_sessions:
        # Очищаємо історію та шаблон, але залишаємо профіль (щоб не шукати в базі знову)
        user_sessions[user_id]['history'] = []
        user_sessions[user_id]['active_template'] = None

        print(f"⌛ Сесію користувача {user_id} закрито через неактивність.")

        # Надсилаємо прощальне повідомлення
        try:
            await context.bot.send_message(
                chat_id=chat_id,
                text="⏳ Час сесії вичерпано (1 година бездіяльності).\nДякую за взаємодію! Історія діалогу очищена.\nЯкщо знадобиться нова заява — просто напишіть мені."
            )
        except Exception as e:
            print(f"Не вдалося надіслати повідомлення про тайм-аут: {e}")


def reset_timeout_timer(user_id, chat_id, context: ContextTypes.DEFAULT_TYPE):
    """Скидає старий таймер і запускає новий на 1 годину"""
    # 1. Видаляємо старі завдання для цього юзера
    current_jobs = context.job_queue.get_jobs_by_name(str(user_id))
    for job in current_jobs:
        job.schedule_removal()

    # 2. Ставимо нове завдання
    context.job_queue.run_once(
        session_timeout_callback,
        4800,  # 4800 секунд = 1 академічна пара
        chat_id=chat_id,
        name=str(user_id),
        data=user_id
    )
